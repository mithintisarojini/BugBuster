Example: Palindrome Check

string s = "madam";
int i = 0, j = s.length()-1;
bool isPal = true;

while(i < j) {
    if(s[i] != s[j]) {
        isPal = false;
        break;
    }
    i++; j--;
}

cout << (isPal ? "Palindrome" : "Not Palindrome");
